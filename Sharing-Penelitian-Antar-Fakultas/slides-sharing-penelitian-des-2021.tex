\section{Motivasi}
\begin{frame}{Motivasi}
\begin{itemize}
	\item<2-> Salah satu indikator mutu dari Program Studi (Prodi) adalah \textbf{jumlah penelitian gabungan antar disiplin ilmu (antar Fakultas)}.
	\item<3-> Saat ini jumlah penelitian antar Fakultas di Prodi Teknik Informatika \textit{minim}.
\end{itemize}
\end{frame}

\section{Sharing \#1: Sumber Daya}
\begin{frame}{Mengidentifikasi Sumber Daya di Universitas}
	\textbf{Contoh}:
	\begin{itemize}
		\item<2-> Mengidentifikasi sumber daya perpustakaan \citep{bunyamin2017automatic,bunyamin2019topic}.
		\item<3-> Mengidentifikasi sumber daya Fakultas Kedokteran \citep{bunyamin2021breast}.
	\end{itemize}
\end{frame}

\section{Sharing \#2: Relasi}
\begin{frame}{Berelasi dengan Fakultas Lain (1/2)}
	\onslide<2->{\textbf{Relasi dengan Fakultas Ekonomi}:}
	\begin{itemize}
		\item<3-> Menggunakan algoritma Machine Learning untuk bidang Ekonomi, khususnya mengkalkulasi hubungan antara \textit{country risk} dan \textit{company performance} \citep{bunyamin2018relationship}.
		\item<4-> Mempelajari ilmu baru, \textit{Time Series Analysis} \& \textit{Prediction} untuk memprediksi Indonesian Economic Growth \citep{Bunyamin2021}.
	\end{itemize}
 
	\onslide<5->{\textbf{Relasi dengan Fakultas Kedokteran}:}
	\begin{itemize}
		\item<6-> Menguji algoritma \textit{image classification} dengan pendekatan \textit{progressive resizing} pada gabungan tiga dataset \textit{breast cancer histopathological images} \citep{bunyamin2021breast}.
	\end{itemize} 		
\end{frame}

\begin{frame}{Berelasi dengan Fakultas Lain (2/2)} 
	\onslide<2->{\textbf{Relasi dengan Fakultas Psikologi \& Perpustakaan}:}
	\begin{itemize}
		\item<3-> Menguji algoritma clustering, yaitu \textit{Latent Dirichlet Allocation} (LDA) with Skip-gram model pada abstrak Tugas Akhir (TA) Fakultas Psikologi \citep{bunyamin2017automatic}.
		\item<4-> Membandingkan algoritma clustering untuk topik-topik TA Fakultas Psikologi \citep{bunyamin2019topic}.
	\end{itemize}	
\end{frame}

\section{Sharing \#3: Humble \& Show Yourself}
\begin{frame}{Bersikap \textit{Humble} \& Show Yourself}
	\textbf{Contoh}:	
	\begin{itemize}
		\item<2-> Melihat \textit{potensi} Fakultas Kedokteran.
		\item<3-> Membuat \textit{janji} dengan ketua LPPM.
		\item<4-> Memberikan \textit{presentasi singkat} tentang skill yang dimiliki.
		\item<5-> Menerima dan mengikuti \textit{pointer} berikutnya.
	\end{itemize}
\end{frame}
